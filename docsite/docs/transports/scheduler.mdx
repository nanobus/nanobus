### Basic configuration

<Tabs
  values={[
    {label: 'YAML', value: 'yaml'},
    {label: 'TypeScript', value: 'typescript'},

  <TabItem value="yaml">

```yaml
transports:
  dapr:
    uses: nanobus.transport.time.scheduler/v1
    with:
      schedules:
        - handler: 'log::helloWorld'
          schedule: '*/1 * * * *'
        - handler: 'log::helloMoon'
          schedule: '*/2 * * * *'
```

  </TabItem>

```typescript
import { SchedulerV1 } from "https://deno.land/x/nanobus_config@v0.0.12/mod.ts";

app.transport(

  SchedulerV1({
    schedules: [{
      handler: log.helloWorld,
      schedule: '*/1 * * * *',
    }, {
      handler: log.helloWorld,
      schedule: '*/2 * * * *',
    }],
  }),
);
```
